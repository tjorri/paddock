# Interactive `HarnessRun` and shell access: implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `spec.mode: Interactive` to `HarnessRun` (long-lived pod accepting multiple prompts over time) and an orthogonal operator-gated shell-access capability, both flowing through new broker-mediated HTTP/WebSocket endpoints. Both features are additive `v1alpha1` evolutions per CLAUDE.md (no `v1alpha2`).

**Architecture:** Three layers of additive change. (1) **CRDs**: `HarnessTemplate.spec.interactive` declares supported mode + lifecycle defaults; `HarnessRun.spec.mode` + `InteractiveOverrides` opts a run in; `HarnessRun.status.interactive` reports runtime state; `BrokerPolicy.grants[*].runs.{interact,shell}` gates access. (2) **Broker**: five new endpoints (`/prompts`, `/stream`, `/interrupt`, `/end`, `/shell`) sharing the existing SA-token auth; new `RenewableProvider` interface called lazily on each prompt to refresh time-bounded credentials (GitHubApp); audit emission via new `AuditKind` values. (3) **Controller + adapter**: mode-aware PodSpec (no `activeDeadlineSeconds`, longer grace, `PADDOCK_INTERACTIVE_MODE` env, NetworkPolicy ingress for broker→adapter), idle/detach/maxLifetime watchdog driving termination, adapter loopback HTTP server with `per-prompt-process` and `persistent-process` mode dispatch.

**Tech Stack:** Go 1.23, controller-runtime, kubebuilder admission webhooks, `nhooyr.io/websocket` (or `gorilla/websocket`) for the broker's WebSocket surface (introduce in Task 13), Ginkgo/Gomega for unit + e2e tests, `controller-runtime/pkg/client/fake` for unit tests, Kind for e2e.

**Spec:** `docs/superpowers/specs/2026-04-29-interactive-harnessrun-design.md`

**Branch:** `docs/interactive-harnessrun` (already created; spec already committed).

type InteractiveSpec struct {

	// template's adapter image supports. The adapter image must declare
	// a matching value via paddock.dev/interactive-modes annotation
	// (validated at pod-spec generation, not at template admission).
	// +kubebuilder:validation:Enum="";per-prompt-process;persistent-process
	// +optional

	// Default 30m.
	// +optional
	// +kubebuilder:default="30m"
	IdleTimeout *metav1.Duration `json:"idleTimeout,omitempty"`

	// attached. Default 15m.
	// +optional
	// +kubebuilder:default="15m"
	DetachIdleTimeout *metav1.Duration `json:"detachIdleTimeout,omitempty"`

	// before termination. Default 5m.
	// +optional
	// +kubebuilder:default="5m"

	// +optional

	MaxLifetime *metav1.Duration `json:"maxLifetime,omitempty"`

	// +optional

// validateInteractiveSpec checks the optional InteractiveSpec for
// internally-consistent timeout bounds. The Mode field's enum is
// enforced by the kubebuilder validation tag at the API layer; here we
// only check cross-field invariants and positive durations.
func validateInteractiveSpec(in *paddockv1alpha1.InteractiveSpec) error {
	if in == nil {
		return nil
	}
	posOrZero := func(name string, d *metav1.Duration) error {
		if d == nil {
			return nil
		}
		if d.Duration <= 0 {
			return fmt.Errorf("interactive.%s must be positive", name)
		}
		return nil
	}
	for name, d := range map[string]*metav1.Duration{
		"idleTimeout":       in.IdleTimeout,
		"detachIdleTimeout": in.DetachIdleTimeout,
		"detachTimeout":     in.DetachTimeout,
		"maxLifetime":       in.MaxLifetime,
	} {
		if err := posOrZero(name, d); err != nil {
			return err
		}
	}
	if in.IdleTimeout != nil && in.MaxLifetime != nil &&
		in.IdleTimeout.Duration > in.MaxLifetime.Duration {
		return fmt.Errorf("interactive.idleTimeout must not exceed maxLifetime")
	}
	if in.DetachTimeout != nil && in.MaxLifetime != nil &&
		in.DetachTimeout.Duration > in.MaxLifetime.Duration {
		return fmt.Errorf("interactive.detachTimeout must not exceed maxLifetime")
	}
	if in.DetachIdleTimeout != nil && in.IdleTimeout != nil &&
		in.DetachIdleTimeout.Duration > in.IdleTimeout.Duration {
		return fmt.Errorf("interactive.detachIdleTimeout must not exceed idleTimeout")
	}
	return nil
}
```

// mergeShell takes the most-permissive union of two ShellCapability
// values. Used when multiple BrokerPolicies grant shell access for
// the same template.
func mergeShell(a, b *paddockv1alpha1.ShellCapability) *paddockv1alpha1.ShellCapability {
	if a == nil {
		return b.DeepCopy()
	}
	out := a.DeepCopy()
	// Target: agent > adapter (more permissive).
	if b.Target == "agent" {
		out.Target = "agent"
	}
	// Command: keep first non-empty.
	if len(out.Command) == 0 && len(b.Command) > 0 {
		out.Command = append([]string{}, b.Command...)
	}
	// AllowedPhases: union (empty in either grant means "all phases").
	if len(out.AllowedPhases) == 0 || len(b.AllowedPhases) == 0 {
		out.AllowedPhases = nil
	} else {
		seen := map[paddockv1alpha1.HarnessRunPhase]struct{}{}
		for _, p := range out.AllowedPhases {
			seen[p] = struct{}{}
		}
		for _, p := range b.AllowedPhases {
			if _, ok := seen[p]; !ok {
				out.AllowedPhases = append(out.AllowedPhases, p)
				seen[p] = struct{}{}
			}
		}
	}
	// RecordTranscript: any-grant-true wins.
	if b.RecordTranscript {
		out.RecordTranscript = true
	}
	return out
}

package providers

import (
	"context"

	paddockv1alpha1 "paddock.dev/paddock/api/v1alpha1"
)

// RenewableProvider is an optional companion interface to Provider for
// providers whose issued credentials have a finite lifetime. Broker
// calls Renew lazily on each prompt submission for any lease whose
// ExpiresAt is within a configurable window (default 5 min).
//
// Implementing Renew is opt-in. Providers without expiry (Static,
// AnthropicAPI, UserSuppliedSecret, PATPool) leave Renew unimplemented;
// the broker uses RenewableProviderOf below to detect support.
type RenewableProvider interface {
	Provider

	// Renew re-issues a lease without revoking the prior one's identity.
	// The returned IssueResult carries a fresh Value and (typically) a
	// later ExpiresAt; the broker atomically swaps it into the in-memory
	// lease registry and patches HarnessRun.status.issuedLeases.
	//
	// Failure is non-fatal: the broker emits an audit event of kind
	// credential-renewal-failed, leaves the existing lease in place,
	// and continues. The agent sees the same env it had — at worst the
	// upstream eventually 401s and surfaces as a PaddockEvent: Error.
	Renew(ctx context.Context, lease paddockv1alpha1.IssuedLease) (*IssueResult, error)
}

// RenewableProviderOf returns the provider as a RenewableProvider if it
// implements the interface, else nil. Use type assertion at call sites
// instead of a registry to keep providers self-contained.
func RenewableProviderOf(p Provider) RenewableProvider {
	rp, _ := p.(RenewableProvider)
	return rp
}

package broker

import (
	"context"
	"time"

	paddockv1alpha1 "paddock.dev/paddock/api/v1alpha1"
	"paddock.dev/paddock/internal/broker/providers"
	"sigs.k8s.io/controller-runtime/pkg/log"
)

// RenewalWalker performs lazy renewal of time-bounded credential leases.
// Called by the prompt handler on each /v1/runs/.../prompts request.
type RenewalWalker struct {
	registry map[string]providers.Provider
	window   time.Duration
	audit    *AuditWriter
}

// NewRenewalWalker constructs a walker. window is the lookahead from now
// — leases expiring within window are renewed.
func NewRenewalWalker(registry map[string]providers.Provider, window time.Duration, audit *AuditWriter) *RenewalWalker {
	return &RenewalWalker{registry: registry, window: window, audit: audit}
}

// WalkAndRenew iterates the run's leases, calls Renew on any provider
// that supports it for leases expiring within the configured window,
// and returns an updated leases slice. Renewal failures are non-fatal:
// the original lease is preserved and an audit event is emitted.
//
// Caller is responsible for persisting the updated slice into
// HarnessRun.status.issuedLeases.
func (w *RenewalWalker) WalkAndRenew(ctx context.Context, namespace, runName string, leases []paddockv1alpha1.IssuedLease) ([]paddockv1alpha1.IssuedLease, error) {
	logger := log.FromContext(ctx).WithValues("run", runName, "namespace", namespace)
	out := append([]paddockv1alpha1.IssuedLease{}, leases...)
	for i, lease := range out {
		if lease.ExpiresAt == nil {
			continue
		}
		if time.Until(lease.ExpiresAt.Time) > w.window {
			continue
		}
		p, ok := w.registry[lease.Provider]
		if !ok {
			continue
		}
		rp := providers.RenewableProviderOf(p)
		if rp == nil {
			continue // static provider — slot stays bound for run lifetime
		}
		newRes, err := rp.Renew(ctx, lease)
		if err != nil {
			logger.Error(err, "renew", "provider", lease.Provider, "leaseID", lease.LeaseID)
			if w.audit != nil {
				w.audit.CredentialRenewalFailed(ctx, namespace, runName, lease.Provider, lease.LeaseID, err)
			}
			continue
		}
		out[i].ExpiresAt = newRes.ExpiresAt
		// Note: the actual token *value* is held in the broker's
		// in-memory lease registry, not in IssuedLease (which only
		// carries identity / TTL metadata). The caller is responsible
		// for swapping in-memory state — the walker only updates the
		// status-level expiry record.
		if w.audit != nil {
			w.audit.CredentialRenewed(ctx, namespace, runName, lease.Provider, lease.LeaseID, newRes.ExpiresAt)
		}
	}
	return out, nil
}

/*

package main

import (
	"context"
	"encoding/json"
	"net"
	"net/http"
)

// Driver is the interface a mode-specific implementation must satisfy.
// per-prompt-process and persistent-process implementations live in
// per_prompt.go and persistent.go respectively.
type Driver interface {
	// SubmitPrompt accepts a prompt and dispatches it to the mode's
	// handler. Synchronous-or-asynchronous is implementation-specific;
	// the server returns 202 Accepted regardless of completion.
	SubmitPrompt(ctx context.Context, p Prompt) error

	// Interrupt cancels the in-flight turn without ending the run.
	Interrupt(ctx context.Context) error

	// End cleanly shuts down the run (terminates the agent process).
	End(ctx context.Context) error

	// StreamHandler returns the WebSocket handler for /stream. In
	// per-prompt-process mode this is typically a no-op; in
	// persistent-process mode it pumps stdin to the long-lived agent
	// and demuxes events back.
	StreamHandler() http.Handler
}

// Prompt is the wire shape received on POST /prompts.
type Prompt struct {
	Text    string `json:"text"`
	Seq     int32  `json:"seq"`
	Submitter string `json:"submitter,omitempty"`
}

// Config configures the loopback HTTP server.
type Config struct {
	Mode   string // "per-prompt-process" | "persistent-process"
	Driver Driver
}

// Server is the loopback HTTP server exposing /prompts, /interrupt,
// /end, /stream to the broker. Listens only on 127.0.0.1.
type Server struct {
	cfg  Config
	mux  *http.ServeMux
	httpServer *http.Server
}

// NewServer constructs a Server.
func NewServer(cfg Config) *Server {
	s := &Server{cfg: cfg, mux: http.NewServeMux()}
	s.mux.HandleFunc("/prompts", s.handlePrompts)
	s.mux.HandleFunc("/interrupt", s.handleInterrupt)
	s.mux.HandleFunc("/end", s.handleEnd)
	s.mux.Handle("/stream", cfg.Driver.StreamHandler())
	return s
}

// Handler returns the underlying mux for tests.
func (s *Server) Handler() http.Handler { return s.mux }

// Listen binds a loopback TCP listener at the given address.
func (s *Server) Listen(addr string) (net.Listener, error) {
	return net.Listen("tcp", addr)
}

// Serve runs the HTTP server on the listener until Shutdown is called.
func (s *Server) Serve(ln net.Listener) error {
	s.httpServer = &http.Server{Handler: s.mux}
	if err := s.httpServer.Serve(ln); err != nil && err != http.ErrServerClosed {
		return err
	}
	return nil
}

// Shutdown gracefully shuts the server down.
func (s *Server) Shutdown(ctx context.Context) error {
	if s.httpServer == nil {
		return nil
	}
	return s.httpServer.Shutdown(ctx)
}

func (s *Server) handlePrompts(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		http.Error(w, "POST only", http.StatusMethodNotAllowed)
		return
	}
	var p Prompt
	if err := json.NewDecoder(r.Body).Decode(&p); err != nil {
		http.Error(w, "invalid body: "+err.Error(), http.StatusBadRequest)
		return
	}
	if err := s.cfg.Driver.SubmitPrompt(r.Context(), p); err != nil {
		http.Error(w, "submit: "+err.Error(), http.StatusInternalServerError)
		return
	}
	w.WriteHeader(http.StatusAccepted)
}

func (s *Server) handleInterrupt(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		http.Error(w, "POST only", http.StatusMethodNotAllowed)
		return
	}
	if err := s.cfg.Driver.Interrupt(r.Context()); err != nil {
		http.Error(w, "interrupt: "+err.Error(), http.StatusInternalServerError)
		return
	}
	w.WriteHeader(http.StatusAccepted)
}

func (s *Server) handleEnd(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		http.Error(w, "POST only", http.StatusMethodNotAllowed)
		return
	}
	if err := s.cfg.Driver.End(r.Context()); err != nil {
		http.Error(w, "end: "+err.Error(), http.StatusInternalServerError)
		return
	}
	w.WriteHeader(http.StatusAccepted)
}

// perPromptDriver implements Driver for the per-prompt-process mode.
// Each SubmitPrompt spawns a fresh agent invocation against the
// workspace, with a per-turn prompt file and the harness's session-resume
// flag (e.g. `claude --resume <id>`) so the agent sees prior context
// from the workspace + harness session metadata.
type perPromptDriver struct {
	logger     *slog.Logger
	workspace  string
	runName    string
	claudeBin  string
	mu         sync.Mutex
	currentCmd *exec.Cmd
}

// NewPerPromptDriver constructs a per-prompt-process Driver. Reads
// PADDOCK_WORKSPACE, PADDOCK_RUN_NAME, PADDOCK_CLAUDE_BINARY (defaults
// to "claude") from env.
func NewPerPromptDriver(logger *slog.Logger) Driver {
	bin := os.Getenv("PADDOCK_CLAUDE_BINARY")
	if bin == "" {
		bin = "claude"
	}
	return &perPromptDriver{
		logger:    logger,
		workspace: os.Getenv("PADDOCK_WORKSPACE"),
		runName:   os.Getenv("PADDOCK_RUN_NAME"),
		claudeBin: bin,
	}
}

func (d *perPromptDriver) SubmitPrompt(ctx context.Context, p Prompt) error {
	dir := filepath.Join(d.workspace, ".paddock", "runs", d.runName, "prompts")
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return fmt.Errorf("mkdir prompts: %w", err)
	}
	promptPath := filepath.Join(dir, fmt.Sprintf("%d.txt", p.Seq))
	if err := os.WriteFile(promptPath, []byte(p.Text), 0o600); err != nil {
		return fmt.Errorf("write prompt: %w", err)
	}

	d.mu.Lock()
	if d.currentCmd != nil {
		d.mu.Unlock()
		return fmt.Errorf("a prompt is already in flight")
	}
	args := []string{"--print", "--input-format", "stream-json", "--output-format", "stream-json"}
	cmd := exec.CommandContext(ctx, d.claudeBin, args...)
	cmd.Dir = d.workspace
	cmd.SysProcAttr = &syscall.SysProcAttr{Setpgid: true} // for clean SIGINT to whole group
	stdin, err := cmd.StdinPipe()
	if err != nil {
		d.mu.Unlock()
		return fmt.Errorf("stdin pipe: %w", err)
	}
	stdout, err := cmd.StdoutPipe()
	if err != nil {
		d.mu.Unlock()
		return fmt.Errorf("stdout pipe: %w", err)
	}
	if err := cmd.Start(); err != nil {
		d.mu.Unlock()
		return fmt.Errorf("start: %w", err)
	}
	d.currentCmd = cmd
	d.mu.Unlock()

	go func() {
		defer func() {
			d.mu.Lock()
			d.currentCmd = nil
			d.mu.Unlock()
		}()
		// Feed the prompt as a single user message in stream-json shape.
		_, _ = fmt.Fprintf(stdin, `{"type":"user","message":{"content":[{"type":"text","text":%q}]}}`+"\n", p.Text)
		_ = stdin.Close()
		// Pipe stdout to the events writer (existing helper from the
		// batch path: appendEventsFromStreamJSON or whatever the
		// adapter's main.go exposes). For tests we don't assert on
		// events.jsonl output — that's covered in Task 14 (e2e).
		_ = drainStreamJSON(d.logger, stdout, d.workspace, d.runName)
		_ = cmd.Wait()
	}()

	return nil
}

func (d *perPromptDriver) Interrupt(ctx context.Context) error {
	d.mu.Lock()
	cmd := d.currentCmd
	d.mu.Unlock()
	if cmd == nil || cmd.Process == nil {
		return nil
	}
	// SIGINT to the whole process group.
	if err := syscall.Kill(-cmd.Process.Pid, syscall.SIGINT); err != nil {
		return fmt.Errorf("signal: %w", err)
	}
	return nil
}

func (d *perPromptDriver) End(ctx context.Context) error {
	// Clean shutdown: send TERM to any in-flight, then exit the adapter
	// process. The adapter's main loop's signal handler will Shutdown the
	// HTTP server; we just nudge the process out.
	d.mu.Lock()
	cmd := d.currentCmd
	d.mu.Unlock()
	if cmd != nil && cmd.Process != nil {
		_ = syscall.Kill(-cmd.Process.Pid, syscall.SIGTERM)
	}
	go func() {
		// Allow the HTTP response to flush before exiting.
		// 200 ms is plenty.
		// (In practice the broker initiates End → adapter dies → controller
		// observes pod termination → run reaches Cancelled. The 200 ms is a
		// margin, not a contract.)
		// time.Sleep would be wrong here without an import; see actual code.
		os.Exit(0)
	}()
	return nil
}

func (d *perPromptDriver) StreamHandler() http.Handler {
	// per-prompt-process doesn't expose a stream — events flow via
	// events.jsonl on the PVC (existing mechanism).
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		http.Error(w, "stream not supported in per-prompt-process mode", http.StatusBadRequest)
	})
}

// drainStreamJSON reads stream-json lines from r and appends each one
// (after conversion through convertLine) to the run's events.jsonl on
// the PVC. Same convertLine helper the batch tail loop uses (defined
// in cmd/adapter-claude-code/convert.go).
func drainStreamJSON(logger *slog.Logger, r io.Reader, workspace, runName string) error {
	eventsPath := filepath.Join(workspace, ".paddock", "runs", runName, "events.jsonl")
	if err := os.MkdirAll(filepath.Dir(eventsPath), 0o755); err != nil {
		return fmt.Errorf("mkdir events: %w", err)
	}
	out, err := os.OpenFile(eventsPath, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0o600)
	if err != nil {
		return fmt.Errorf("open events: %w", err)
	}
	defer out.Close()
	br := bufio.NewReader(r)
	enc := json.NewEncoder(out)
	for {
		line, err := br.ReadBytes('\n')
		if len(line) > 0 {
			events, cerr := convertLine(line)
			if cerr != nil {
				logger.Warn("convertLine", "err", cerr)
			}
			for _, ev := range events {
				if werr := enc.Encode(ev); werr != nil {
					return fmt.Errorf("write event: %w", werr)
				}
			}
		}
		if err == io.EOF {
			return nil
		}
		if err != nil {
			return fmt.Errorf("read: %w", err)
		}
	}
}

package broker

import (
	"context"
	"fmt"
	"io"
	"net/http"
	"sync"
)

// AdapterResolver looks up the loopback HTTP address (host:port) of the
// adapter sidecar for a given (namespace, run) pair. The broker server
// wires this from the controller-runtime cache via a Pod-IP lookup
// keyed on the HarnessRun's owner Job and Pod label.
type AdapterResolver func(ctx context.Context, namespace, runName string) (string, error)

// InteractiveRouter keeps per-run state (attach counter, turn sequence)
// and forwards prompt/interrupt/end calls to the adapter's loopback
// HTTP server.
type InteractiveRouter struct {
	resolve AdapterResolver
	client  *http.Client

	mu    sync.Mutex
	state map[string]*runState
}

type runState struct {
	attached int32
	turnSeq  int32
}

// NewInteractiveRouter constructs an InteractiveRouter. Resolver may be
// nil for tests that exercise only the in-memory counters.
func NewInteractiveRouter(resolver AdapterResolver) *InteractiveRouter {
	return &InteractiveRouter{
		resolve: resolver,
		client:  &http.Client{},
		state:   map[string]*runState{},
	}
}

func (r *InteractiveRouter) get(namespace, runName string) *runState {
	r.mu.Lock()
	defer r.mu.Unlock()
	key := namespace + "/" + runName
	if s, ok := r.state[key]; ok {
		return s
	}
	s := &runState{}
	r.state[key] = s
	return s
}

// OnAttach increments the attach counter for the run.
func (r *InteractiveRouter) OnAttach(namespace, runName string) {
	s := r.get(namespace, runName)
	r.mu.Lock()
	s.attached++
	r.mu.Unlock()
}

// OnDetach decrements the attach counter for the run.
func (r *InteractiveRouter) OnDetach(namespace, runName string) {
	s := r.get(namespace, runName)
	r.mu.Lock()
	if s.attached > 0 {
		s.attached--
	}
	r.mu.Unlock()
}

// AttachedCount returns the current attach count.
func (r *InteractiveRouter) AttachedCount(namespace, runName string) int32 {
	s := r.get(namespace, runName)
	r.mu.Lock()
	defer r.mu.Unlock()
	return s.attached
}

// NextTurnSeq returns the next turn sequence number for the run,
// monotonically incrementing.
func (r *InteractiveRouter) NextTurnSeq(namespace, runName string) int32 {
	s := r.get(namespace, runName)
	r.mu.Lock()
	defer r.mu.Unlock()
	s.turnSeq++
	return s.turnSeq
}

// ForgetRun drops cached state for a terminated run.
func (r *InteractiveRouter) ForgetRun(namespace, runName string) {
	r.mu.Lock()
	defer r.mu.Unlock()
	delete(r.state, namespace+"/"+runName)
}

// ForwardPrompt forwards the request body to the adapter's /prompts
// endpoint and copies the adapter's response back to w.
func (r *InteractiveRouter) ForwardPrompt(ctx context.Context, w http.ResponseWriter, req *http.Request, namespace, runName string) {
	r.forward(ctx, w, req, namespace, runName, "/prompts")
}

// ForwardInterrupt forwards to the adapter's /interrupt.
func (r *InteractiveRouter) ForwardInterrupt(ctx context.Context, w http.ResponseWriter, req *http.Request, namespace, runName string) {
	r.forward(ctx, w, req, namespace, runName, "/interrupt")
}

// ForwardEnd forwards to the adapter's /end.
func (r *InteractiveRouter) ForwardEnd(ctx context.Context, w http.ResponseWriter, req *http.Request, namespace, runName string) {
	r.forward(ctx, w, req, namespace, runName, "/end")
}

func (r *InteractiveRouter) forward(ctx context.Context, w http.ResponseWriter, req *http.Request, namespace, runName, path string) {
	addr, err := r.resolve(ctx, namespace, runName)
	if err != nil {
		http.Error(w, fmt.Sprintf("resolve adapter: %v", err), http.StatusBadGateway)
		return
	}
	u := fmt.Sprintf("http://%s%s", addr, path)
	out, err := http.NewRequestWithContext(ctx, req.Method, u, req.Body)
	if err != nil {
		http.Error(w, fmt.Sprintf("build request: %v", err), http.StatusInternalServerError)
		return
	}
	for k, vs := range req.Header {
		for _, v := range vs {
			out.Header.Add(k, v)
		}
	}
	resp, err := r.client.Do(out)
	if err != nil {
		http.Error(w, fmt.Sprintf("forward: %v", err), http.StatusBadGateway)
		return
	}
	defer resp.Body.Close()
	for k, vs := range resp.Header {
		for _, v := range vs {
			w.Header().Add(k, v)
		}
	}
	w.WriteHeader(resp.StatusCode)
	_, _ = io.Copy(w, resp.Body)
}
```

// handlePrompts forwards a prompt to the adapter after admission +
// renewal walk + audit emission.
func (s *Server) handlePrompts(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		http.Error(w, "POST only", http.StatusMethodNotAllowed)
		return
	}
	ctx := r.Context()
	caller, err := s.authenticate(ctx, r)
	if err != nil {
		http.Error(w, err.Error(), http.StatusUnauthorized)
		return
	}
	ns, runName, err := pathRunIdentity(r)
	if err != nil {
		http.Error(w, err.Error(), http.StatusBadRequest)
		return
	}
	// Cross-namespace: caller must be in the target namespace.
	if caller.Namespace != ns {
		http.Error(w, "namespace mismatch", http.StatusForbidden)
		return
	}

	var run paddockv1alpha1.HarnessRun
	if err := s.client.Get(ctx, client.ObjectKey{Namespace: ns, Name: runName}, &run); err != nil {
		http.Error(w, "run not found", http.StatusNotFound)
		return
	}

	// Admission: spec.mode == Interactive AND policy grants runs.interact.
	if run.Spec.Mode != paddockv1alpha1.HarnessRunModeInteractive {
		http.Error(w, "run is not Interactive mode", http.StatusBadRequest)
		return
	}
	if !s.allowInteract(ctx, &run) {
		http.Error(w, "no BrokerPolicy grants runs.interact for this run's template", http.StatusForbidden)
		return
	}

	// Phase guard: only Idle accepts new prompts; Running already has one in flight.
	if run.Status.Phase == paddockv1alpha1.HarnessRunPhaseRunning {
		http.Error(w, "a prompt is already in flight", http.StatusConflict)
		return
	}

	// Read body.
	var body brokerapi.PromptRequest
	if err := json.NewDecoder(r.Body).Decode(&body); err != nil {
		http.Error(w, "invalid body: "+err.Error(), http.StatusBadRequest)
		return
	}
	if len(body.Text) == 0 || len(body.Text) > paddockv1alpha1.MaxInlinePromptBytes {
		http.Error(w, "prompt empty or too large", http.StatusBadRequest)
		return
	}

	// Renewal walk.
	updated, _ := s.renewer.WalkAndRenew(ctx, ns, runName, run.Status.IssuedLeases)
	if !equalLeases(updated, run.Status.IssuedLeases) {
		_ = s.patchIssuedLeases(ctx, &run, updated)
	}

	// Allocate turn sequence.
	seq := s.router.NextTurnSeq(ns, runName)

	// Audit.
	hash := sha256.Sum256([]byte(body.Text))
	s.audit.write(ctx, auditing.NewPromptSubmitted(auditing.PromptAuditInput{
		RunName:      runName,
		Namespace:    ns,
		SubmitterSA:  caller.ServiceAccount,
		PromptHash:   "sha256:" + hex.EncodeToString(hash[:]),
		PromptLength: len(body.Text),
		TurnSeq:      seq,
	}))

	// Forward to adapter. The adapter expects {text, seq, submitter}.
	forwardBody, _ := json.Marshal(struct {
		Text      string `json:"text"`
		Seq       int32  `json:"seq"`
		Submitter string `json:"submitter"`
	}{Text: body.Text, Seq: seq, Submitter: caller.ServiceAccount})
	r.Body = newReadCloser(forwardBody)
	r.ContentLength = int64(len(forwardBody))
	s.router.ForwardPrompt(ctx, w, r, ns, runName)

	// Patch status.interactive on success (best-effort).
	now := nowMeta()
	if run.Status.Interactive == nil {
		run.Status.Interactive = &paddockv1alpha1.InteractiveStatus{}
	}
	run.Status.Interactive.PromptCount++
	run.Status.Interactive.LastPromptAt = &now
	run.Status.Interactive.CurrentTurnSeq = &seq
	run.Status.Interactive.IdleSince = nil
	_ = s.client.Status().Update(ctx, &run)
}

package controller

import (
	"time"

	paddockv1alpha1 "paddock.dev/paddock/api/v1alpha1"
)

type watchdogAction int

const (
	watchdogActionNone watchdogAction = iota
	watchdogActionCancelIdle
	watchdogActionCancelDetach
	watchdogActionCancelMaxLifetime
)

func (a watchdogAction) Reason() string {
	switch a {
	case watchdogActionCancelIdle:
		return "idle"
	case watchdogActionCancelDetach:
		return "detach"
	case watchdogActionCancelMaxLifetime:
		return "max-lifetime"
	}
	return ""
}

// nextDeadline returns the watchdog action to take and (when action is
// none) how long to requeue before the next check. Pure function.
func nextDeadline(now time.Time, run *paddockv1alpha1.HarnessRun, tpl *resolvedTemplate) (watchdogAction, time.Duration) {
	if run.Spec.Mode != paddockv1alpha1.HarnessRunModeInteractive {
		return watchdogActionNone, 0
	}
	if tpl.Interactive == nil {
		return watchdogActionNone, 0
	}
	istat := run.Status.Interactive
	if istat == nil {
		istat = &paddockv1alpha1.InteractiveStatus{}
	}

	bound := func(d *metav1.Duration, fallback time.Duration) time.Duration {
		if d == nil {
			return fallback
		}
		return d.Duration
	}
	idleTimeout := bound(tpl.Interactive.IdleTimeout, 30*time.Minute)
	detachIdleTimeout := bound(tpl.Interactive.DetachIdleTimeout, 15*time.Minute)
	detachTimeout := bound(tpl.Interactive.DetachTimeout, 5*time.Minute)
	maxLifetime := bound(tpl.Interactive.MaxLifetime, 24*time.Hour)
	// Apply per-run overrides (already validated to not exceed template
	// bounds at admission, plus webhook check).
	if ov := run.Spec.InteractiveOverrides; ov != nil {
		if ov.IdleTimeout != nil {
			idleTimeout = ov.IdleTimeout.Duration
		}
		if ov.DetachIdleTimeout != nil {
			detachIdleTimeout = ov.DetachIdleTimeout.Duration
		}
		if ov.DetachTimeout != nil {
			detachTimeout = ov.DetachTimeout.Duration
		}
		if ov.MaxLifetime != nil {
			maxLifetime = ov.MaxLifetime.Duration
		}
	}

	// Pick effective idle timeout based on attach state.
	effIdle := idleTimeout
	if istat.AttachedSessions == 0 {
		effIdle = detachIdleTimeout
	}

	// Time since each axis.
	var sinceIdle, sinceDetach, sinceCreate time.Duration
	if istat.IdleSince != nil {
		sinceIdle = now.Sub(istat.IdleSince.Time)
	}
	if istat.AttachedSessions == 0 && istat.LastAttachedAt != nil {
		sinceDetach = now.Sub(istat.LastAttachedAt.Time)
	}
	sinceCreate = now.Sub(run.CreationTimestamp.Time)

	if istat.IdleSince != nil && sinceIdle >= effIdle {
		return watchdogActionCancelIdle, 0
	}
	if istat.AttachedSessions == 0 && istat.LastAttachedAt != nil && sinceDetach >= detachTimeout {
		return watchdogActionCancelDetach, 0
	}
	if sinceCreate >= maxLifetime {
		return watchdogActionCancelMaxLifetime, 0
	}

	// None fires; compute the smallest remaining for requeue.
	candidates := []time.Duration{}
	if istat.IdleSince != nil {
		candidates = append(candidates, effIdle-sinceIdle)
	}
	if istat.AttachedSessions == 0 && istat.LastAttachedAt != nil {
		candidates = append(candidates, detachTimeout-sinceDetach)
	}
	candidates = append(candidates, maxLifetime-sinceCreate)
	smallest := candidates[0]
	for _, c := range candidates {
		if c < smallest {
			smallest = c
		}
	}
	if smallest < time.Second {
		smallest = time.Second
	}
	return watchdogActionNone, smallest
}

type persistentDriver struct {
	logger *slog.Logger
	mu     sync.Mutex
	cmd    *exec.Cmd
	stdin  io.WriteCloser
	stdout io.ReadCloser
	// fanout broadcasts each parsed stream-json event line to all
	// attached WebSocket clients.
	fanout *eventFanout

// NewPersistentDriver constructs a persistent-process Driver.
func NewPersistentDriver(logger *slog.Logger) Driver {
	d := &persistentDriver{logger: logger, fanout: newEventFanout()}
	if err := d.start(); err != nil {
		logger.Error("start persistent agent", "err", err)
		// Mark broken state; SubmitPrompt will return error.
	}
	return d
}

func (d *persistentDriver) start() error {
	bin := os.Getenv("PADDOCK_CLAUDE_BINARY")
	if bin == "" {
		bin = "claude"
	}
	cmd := exec.Command(bin, "--input-format", "stream-json", "--output-format", "stream-json")
	cmd.Dir = os.Getenv("PADDOCK_WORKSPACE")
	stdin, err := cmd.StdinPipe()
	if err != nil {
		return err
	}
	stdout, err := cmd.StdoutPipe()
	if err != nil {
		return err
	}
	if err := cmd.Start(); err != nil {
		return err
	}
	d.cmd = cmd
	d.stdin = stdin
	d.stdout = stdout
	go d.demux()
	return nil
}

func (d *persistentDriver) demux() {
	br := bufio.NewReader(d.stdout)
	for {
		line, err := br.ReadBytes('\n')
		if len(line) > 0 {
			d.fanout.broadcast(line)
			// Also append to events.jsonl on PVC for resume.
			// (Use the existing convertLine + writer helper from main.go.)
		}
		if err != nil {
			d.logger.Info("agent stdout closed", "err", err)
			return
		}
	}
}

func (d *persistentDriver) SubmitPrompt(ctx context.Context, p Prompt) error {
	d.mu.Lock()
	defer d.mu.Unlock()
	if d.stdin == nil {
		return fmt.Errorf("agent not running")
	}
	msg := struct {
		Type    string `json:"type"`
		Message struct {
			Content []struct {
				Type string `json:"type"`
				Text string `json:"text"`
			} `json:"content"`
		} `json:"message"`
		Seq int32 `json:"_paddock_seq"`
	}{Type: "user", Seq: p.Seq}
	msg.Message.Content = append(msg.Message.Content, struct {
		Type string `json:"type"`
		Text string `json:"text"`
	}{Type: "text", Text: p.Text})
	b, _ := json.Marshal(msg)
	if _, err := d.stdin.Write(append(b, '\n')); err != nil {
		return err
	}
	return nil
}

func (d *persistentDriver) Interrupt(ctx context.Context) error {
	d.mu.Lock()
	defer d.mu.Unlock()
	if d.cmd == nil || d.cmd.Process == nil {
		return nil
	}
	// SIGINT to the agent; stream-json driver handles it as turn cancel.
	return d.cmd.Process.Signal(os.Interrupt)
}

func (d *persistentDriver) End(ctx context.Context) error {
	d.mu.Lock()
	if d.stdin != nil {
		_ = d.stdin.Close()
	}
	cmd := d.cmd
	d.mu.Unlock()
	if cmd != nil {
		_ = cmd.Wait()
	}
	go func() { os.Exit(0) }()
	return nil
}

func (d *persistentDriver) StreamHandler() http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		c, err := websocket.Accept(w, r, &websocket.AcceptOptions{
			Subprotocols: []string{"paddock.stream.v1"},
		})
		if err != nil {
			return
		}
		defer c.Close(websocket.StatusInternalError, "exit")

		// Subscribe to the fanout.
		ch := d.fanout.subscribe()
		defer d.fanout.unsubscribe(ch)

		ctx := r.Context()
		// Outbound: drain the fanout to the client.
		go func() {
			for line := range ch {
				if err := c.Write(ctx, websocket.MessageText, line); err != nil {
					return
				}
			}
		}()
		// Inbound: write client messages to agent stdin (stream-json control).
		for {
			_, msg, err := c.Read(ctx)
			if err != nil {
				return
			}
			d.mu.Lock()
			stdin := d.stdin
			d.mu.Unlock()
			if stdin == nil {
				return
			}
			if _, err := stdin.Write(append(msg, '\n')); err != nil {
				return
			}
		}
	})
}

// eventFanout broadcasts each line to all subscribed channels.
type eventFanout struct {
	mu   sync.Mutex
	subs map[chan []byte]struct{}
}

func newEventFanout() *eventFanout { return &eventFanout{subs: map[chan []byte]struct{}{}} }

func (f *eventFanout) subscribe() chan []byte {
	ch := make(chan []byte, 64)
	f.mu.Lock()
	f.subs[ch] = struct{}{}
	f.mu.Unlock()
	return ch
}

func (f *eventFanout) unsubscribe(ch chan []byte) {
	f.mu.Lock()
	delete(f.subs, ch)
	f.mu.Unlock()
	close(ch)
}

func (f *eventFanout) broadcast(line []byte) {
	cp := append([]byte{}, line...)
	f.mu.Lock()
	for ch := range f.subs {
		select {
		case ch <- cp:
		default:
			// drop on slow consumer
		}
	}
	f.mu.Unlock()
}

package broker

import (
	"context"
	"net/http"

	paddockv1alpha1 "paddock.dev/paddock/api/v1alpha1"
	"nhooyr.io/websocket"
	"sigs.k8s.io/controller-runtime/pkg/client"
)

func (s *Server) handleStream(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	caller, err := s.authenticate(ctx, r)
	if err != nil {
		http.Error(w, err.Error(), http.StatusUnauthorized)
		return
	}
	ns, runName, err := pathRunIdentity(r)
	if err != nil {
		http.Error(w, err.Error(), http.StatusBadRequest)
		return
	}
	if caller.Namespace != ns {
		http.Error(w, "namespace mismatch", http.StatusForbidden)
		return
	}
	var run paddockv1alpha1.HarnessRun
	if err := s.client.Get(ctx, client.ObjectKey{Namespace: ns, Name: runName}, &run); err != nil {
		http.Error(w, "run not found", http.StatusNotFound)
		return
	}
	if !s.allowInteract(ctx, &run) {
		http.Error(w, "no BrokerPolicy grants runs.interact", http.StatusForbidden)
		return
	}

	// Resolve adapter address.
	addr, err := s.router.resolve(ctx, ns, runName)
	if err != nil {
		http.Error(w, "resolve adapter: "+err.Error(), http.StatusBadGateway)
		return
	}

	// Accept the client connection.
	clientConn, err := websocket.Accept(w, r, &websocket.AcceptOptions{
		Subprotocols: []string{"paddock.stream.v1"},
	})
	if err != nil {
		return
	}
	defer clientConn.Close(websocket.StatusInternalError, "exit")

	// Track attach.
	s.router.OnAttach(ns, runName)
	defer s.router.OnDetach(ns, runName)
	s.patchAttachStatus(ctx, ns, runName)
	defer s.patchAttachStatus(ctx, ns, runName)

	// Dial adapter.
	upstream, _, err := websocket.Dial(ctx, "ws://"+addr+"/stream", &websocket.DialOptions{
		Subprotocols: []string{"paddock.stream.v1"},
	})
	if err != nil {
		clientConn.Close(websocket.StatusBadGateway, "adapter dial failed")
		return
	}
	defer upstream.Close(websocket.StatusInternalError, "exit")

	errCh := make(chan error, 2)
	go copyWS(ctx, upstream, clientConn, errCh)
	go copyWS(ctx, clientConn, upstream, errCh)
	<-errCh
}

			return
		}

func (s *Server) handleShell(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	caller, err := s.authenticate(ctx, r)
	if err != nil {
		http.Error(w, err.Error(), http.StatusUnauthorized)
		return
	}
	ns, runName, err := pathRunIdentity(r)
	if err != nil {
		http.Error(w, err.Error(), http.StatusBadRequest)
		return
	}
	if caller.Namespace != ns {
		http.Error(w, "namespace mismatch", http.StatusForbidden)
		return
	}

	var run paddockv1alpha1.HarnessRun
	if err := s.client.Get(ctx, client.ObjectKey{Namespace: ns, Name: runName}, &run); err != nil {
		http.Error(w, "run not found", http.StatusNotFound)
		return
	}
	cap := s.allowShell(ctx, &run)
	if cap == nil {
		http.Error(w, "no BrokerPolicy grants runs.shell", http.StatusForbidden)
		return
	}
	if !shellPhaseAllowed(cap, run.Status.Phase) {
		http.Error(w, "shell not allowed in phase "+string(run.Status.Phase), http.StatusBadRequest)
		return
	}

	// Resolve target pod.
	podName, err := s.resolveRunPod(ctx, ns, runName)
	if err != nil {
		http.Error(w, "pod not found: "+err.Error(), http.StatusNotFound)
		return
	}
	containerName := "agent"
	if cap.Target == "adapter" {
		containerName = "event-adapter" // matches the adapter sidecar's name in pod_spec.go
	}
	cmd := cap.Command
	if len(cmd) == 0 {
		cmd = []string{"/bin/bash"} // /bin/sh fallback handled by exec returning ExitError(127); broker retries with sh
	}

	sessionID := uuid.NewString()
	w.Header().Set(brokerapi.HeaderShellSessionID, sessionID)

	// Emit open audit.
	s.audit.write(ctx, auditing.NewShellSessionOpened(auditing.ShellOpenedInput{
		RunName: runName, Namespace: ns, SessionID: sessionID,
		SubmitterSA: caller.ServiceAccount, Target: cap.Target, Command: cmd,
	}))
	startedAt := time.Now()

	// WebSocket upgrade with the client.
	clientConn, err := websocket.Accept(w, r, &websocket.AcceptOptions{
		Subprotocols: []string{"paddock.shell.v1"},
	})
	if err != nil {
		return
	}
	var byteCount int64
	defer func() {
		clientConn.Close(websocket.StatusNormalClosure, "exit")
		s.audit.write(ctx, auditing.NewShellSessionClosed(auditing.ShellClosedInput{
			RunName:    runName,
			Namespace:  ns,
			SessionID:  sessionID,
			DurationMs: time.Since(startedAt).Milliseconds(),
			ByteCount:  byteCount,
		}))
	}()

	// Build exec via SPDY (k8s pods/exec subresource).
	cs, err := kubernetes.NewForConfig(s.restConfig)
	if err != nil {
		clientConn.Close(websocket.StatusInternalError, "client config")
		return
	}
	req := cs.CoreV1().RESTClient().Post().Resource("pods").Namespace(ns).Name(podName).
		SubResource("exec").VersionedParams(&corev1.PodExecOptions{
		Container: containerName,
		Command:   cmd,
		Stdin:     true, Stdout: true, Stderr: true, TTY: true,
	}, scheme.ParameterCodec)
	exec, err := remotecommand.NewSPDYExecutor(s.restConfig, "POST", req.URL())
	if err != nil {
		clientConn.Close(websocket.StatusInternalError, "spdy: "+err.Error())
		return
	}

	// Stream: client WS <-> exec stdio.
	stdinR, stdinW := io.Pipe()
	stdoutR, stdoutW := io.Pipe()

	go func() {
		// Client → stdin
		for {
			_, msg, err := clientConn.Read(ctx)
			if err != nil {
				_ = stdinW.Close()
				return
			}
			byteCount += int64(len(msg))
			if _, err := stdinW.Write(msg); err != nil {
				return
			}
		}
	}()
	go func() {
		// stdout → client
		buf := make([]byte, 4096)
		for {
			n, err := stdoutR.Read(buf)
			if n > 0 {
				_ = clientConn.Write(ctx, websocket.MessageBinary, buf[:n])
			}
			if err != nil {
				return
			}
		}
	}()

	_ = exec.StreamWithContext(ctx, remotecommand.StreamOptions{
		Stdin:  stdinR,
		Stdout: stdoutW,
		Stderr: stdoutW, // merge stderr into stdout for the WebSocket
		Tty:    true,
	})
}

func shellPhaseAllowed(cap *paddockv1alpha1.ShellCapability, phase paddockv1alpha1.HarnessRunPhase) bool {
	if len(cap.AllowedPhases) == 0 {
		_, ok := phasesWithPod[phase]
		return ok
	}
	for _, p := range cap.AllowedPhases {
		if p == phase {
			return true
		}
	}
	return false
}
