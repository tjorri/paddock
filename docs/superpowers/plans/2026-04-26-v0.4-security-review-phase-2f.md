# v0.4 Security Review — Phase 2f (Per-run intermediate CA) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close audit finding F-18 by issuing per-run (and per-Workspace) intermediate CAs via cert-manager, so the cluster root MITM CA private key never leaves cert-manager's signing path and tenant namespaces never receive copies of the root.

**Architecture:** A new cert-manager `ClusterIssuer` of `kind: CA` references the existing `paddock-proxy-ca` Secret as its signing root. The controller creates a per-run/per-Workspace `Certificate` resource with `isCA: true` directly in the tenant namespace; cert-manager produces the per-run/per-Workspace Secret with intermediate keypair. Agent containers mount the **intermediate cert** (`tls.crt`) as their TLS trust anchor — not the cluster root. Tenant A's agent does not trust leaves signed by tenant B's intermediate.

**Tech Stack:** Go, cert-manager-go API (`github.com/cert-manager/cert-manager/pkg/apis/certmanager/v1`), Kubebuilder controller-runtime, controller-runtime fake client for unit tests, kustomize + Helm for manifest packaging.

**Spec:** `docs/superpowers/specs/2026-04-26-v0.4-security-review-phase-2f-design.md`

**Branch (to create):** `phase-2f-per-run-intermediate-ca` — create BEFORE the first commit, per the project's branch-first workflow memory.

- [ ] **Step 2: Create `config/samples/kyverno-cluster-issuer-restriction.yaml`**

Write this file:

```yaml
# Sample Kyverno policy restricting cert-manager.io/v1 Certificate
# resources that reference the paddock-proxy-ca-issuer ClusterIssuer to
# the paddock-controller ServiceAccount.
#
# Without this restriction, any user/ServiceAccount with cluster-wide
# `certificates.cert-manager.io: create` RBAC could request a CA-signed
# intermediate from the Paddock root and forge leaves trusted by other
# tenants' agents (the F-18 attack shape this phase addresses, in a
# different form). Operators with strict multi-tenant requirements
# should apply this policy.
#
# This is a SAMPLE — not applied by default. F-18 / Phase 2f.
apiVersion: kyverno.io/v1
kind: ClusterPolicy
metadata:
  name: paddock-restrict-proxy-ca-issuer
  annotations:
    policies.kyverno.io/title: Restrict paddock-proxy-ca-issuer Certificate creation
    policies.kyverno.io/category: Multi-Tenancy
    policies.kyverno.io/severity: high
    policies.kyverno.io/subject: Certificate
    policies.kyverno.io/description: >-
      Only the paddock-controller ServiceAccount may create
      cert-manager Certificate resources that reference the
      paddock-proxy-ca-issuer ClusterIssuer. Tenants and other
      operators get cluster-wide MITM forgery if this is not enforced.
spec:
  validationFailureAction: Enforce
  background: false
  rules:
    - name: only-paddock-controller-may-use-issuer
      match:
        any:
          - resources:
              kinds:
                - Certificate
      preconditions:
        all:
          - key: "{{ request.object.spec.issuerRef.kind }}"
            operator: Equals
            value: ClusterIssuer
          - key: "{{ request.object.spec.issuerRef.name }}"
            operator: Equals
            value: paddock-proxy-ca-issuer
      validate:
        message: >-
          Only the paddock-controller ServiceAccount in paddock-system may
          create Certificate resources referencing the paddock-proxy-ca-issuer
          ClusterIssuer. F-18 / Phase 2f.
        deny:
          conditions:
            all:
              - key: "{{ request.userInfo.username }}"
                operator: NotEquals
                value: system:serviceaccount:paddock-system:paddock-controller-manager
```

// the backing per-run Secret directly in the run's namespace; the
// controller never reads or writes the intermediate's private key.
//

func (r *HarnessRunReconciler) ensureProxyTLS(ctx context.Context, run *paddockv1alpha1.HarnessRun) (bool, error) {

// ensureProxyCACertificate is the shared cert-manager Certificate
// upsert logic used by the run path (ensureProxyTLS) and the seed path
// (ensureSeedProxyTLS in workspace_broker.go). Returns
// (created, ready, err) — created=true on the first reconcile pass
// where the Certificate is freshly created.
//
// owner is the parent CR (HarnessRun or Workspace); the Certificate
// gets an OwnerReference to it so cascading delete works. ns is the
// namespace where the Certificate is created (typically the parent's
// own namespace). secretName is the name of the backing Secret
// cert-manager will create. commonName is set on the Certificate
// spec; clusterIssuer names the cert-manager ClusterIssuer that signs
// the intermediate. duration is the per-Certificate validity;
// renewBefore is set when non-zero (zero = no renewal, run-pod path).
//
// F-18 / Phase 2f.
func ensureProxyCACertificate(
	ctx context.Context,
	cli client.Client,
	scheme *runtime.Scheme,
	owner client.Object,
	ns, secretName, commonName, clusterIssuer string,
	duration, renewBefore time.Duration,
	labels map[string]string,
) (created bool, ready bool, err error) {
	cert := &cmapi.Certificate{
		ObjectMeta: metav1.ObjectMeta{
			Name:      secretName,
			Namespace: ns,
			Labels:    labels,
		},
	}
	op, err := controllerutil.CreateOrUpdate(ctx, cli, cert, func() error {
		if err := controllerutil.SetControllerReference(owner, cert, scheme); err != nil {
			return err
		}
		cert.Spec.IsCA = true
		cert.Spec.CommonName = commonName
		cert.Spec.SecretName = secretName
		cert.Spec.Duration = &metav1.Duration{Duration: duration}
		if renewBefore > 0 {
			cert.Spec.RenewBefore = &metav1.Duration{Duration: renewBefore}
		} else {
			cert.Spec.RenewBefore = nil
		}
		cert.Spec.PrivateKey = &cmapi.CertificatePrivateKey{
			Algorithm: cmapi.ECDSAKeyAlgorithm,
			Size:      256,
		}
		cert.Spec.Usages = []cmapi.KeyUsage{
			cmapi.UsageDigitalSignature,
			cmapi.UsageKeyEncipherment,
			cmapi.UsageCertSign,
		}
		cert.Spec.IssuerRef = cmmeta.ObjectReference{
			Kind: "ClusterIssuer",
			Name: clusterIssuer,
		}
		return nil
	})
	if err != nil {
		return false, false, fmt.Errorf("upserting Certificate %s/%s: %w", ns, secretName, err)
	}

	// Re-read to pick up status (CreateOrUpdate doesn't fetch status by default).
	if err := cli.Get(ctx, types.NamespacedName{Name: secretName, Namespace: ns}, cert); err != nil {
		return false, false, fmt.Errorf("re-reading Certificate %s/%s: %w", ns, secretName, err)
	}
	for _, c := range cert.Status.Conditions {
		if c.Type == cmapi.CertificateConditionReady && c.Status == cmmeta.ConditionTrue {
			ready = true
			break
		}
	}
	return op == controllerutil.OperationResultCreated, ready, nil
}
